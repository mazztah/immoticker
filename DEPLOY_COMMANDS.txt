# in der Google Cloud Run Konsole ein (Service → "Bearbeiten und neue Version bereitstellen"
# → Tab "Variablen & Secrets" → "+ Variable hinzufügen": GROQ_API_KEY, XAI_API_KEY,
# ANTHROPIC_API_KEY, FILIP_LINKEDIN_URL, FILIP_XING_URL).

# ============================================================
# 1) LOKAL TESTEN
# ============================================================
python3 -m venv .venv
source .venv/bin/activate          # Windows: .venv\Scripts\activate
pip install -r requirements.txt
uvicorn app:app --host 0.0.0.0 --port 8080 --reload
# -> http://localhost:8080  (liest automatisch die lokale .env dank python-dotenv)

# ============================================================
# 2) GITHUB HOCHLADEN (ohne Secrets — .env & diese Datei sind gitignored)
# ============================================================
git init
git add .
git commit -m "KI-Immo-Terminal: Backend, Feeds, Frontend"
git branch -M main
git remote add origin https://github.com/<DEIN-USERNAME>/<DEIN-REPO>.git
git push -u origin main

# ============================================================
# 3) GOOGLE CLOUD RUN DEPLOYEN
#    (nur der Code wird deployed — Keys trägst du danach in der Konsole ein, siehe oben)
# ============================================================
gcloud auth login
gcloud config set project <DEIN-GCP-PROJEKT-ID>

gcloud run deploy ki-immo-terminal \
  --source . \
  --region europe-west1 \
  --allow-unauthenticated

# Danach in der Cloud Run Konsole (console.cloud.google.com/run):
#   Service "ki-immo-terminal" öffnen → "Bearbeiten und neue Version bereitstellen"
#   → Tab "Variablen & Secrets" → Umgebungsvariablen eintragen → "Bereitstellen"
#
# Empfohlene, noch sicherere Variante für Produktion: Secret Manager statt normaler
# Umgebungsvariablen (im selben Tab als "Secret Reference" statt "Variable" wählbar,
# oder per CLI: gcloud secrets create groq-api-key --data-file=-)

# Nach Deployment zeigt gcloud dir die Service-URL an, z.B.:
#   https://ki-immo-terminal-xxxxxxxxxx-ew.a.run.app
